# Runtime dependencies for FFmpeg GUI.
#
# We deliberately keep this minimal. Only PySide6 is needed at runtime; every
# video operation is delegated to FFmpeg/FFprobe via subprocess. This keeps
# the packaged application small and avoids pulling in native wheels that
# would complicate portable builds.
#
# A version floor is set high enough to guarantee the QtMultimedia APIs we
# rely on (QAudioOutput split from QMediaPlayer, QGraphicsVideoItem transform
# behaviour) but we do not pin an exact version so that bug fixes flow in.
PySide6>=6.5
